## 前置インクリメントと後置インクリメントの違い

### a++(後置インクリメント)
    let a = 0;
    console.log(a)
    a = b = a + 1;
    console.log(a, b)
    // 0
    //1,1

### ++a(前置インクリメント)
        let a = 0;
        console.log(a)
        a = (b = a) + 1;
        console.log(a, b)
        //0
        //1,0
