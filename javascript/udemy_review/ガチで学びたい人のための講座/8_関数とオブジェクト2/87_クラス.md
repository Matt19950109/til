## クラス
- コンストラクター関数をクラス表記で書けるようにしたもの

## シンタックスシュガー
- すでにあるものを簡単に記述できるようにしたもの

class Person {
  constructor() {
    this.name = name;
    this.age = age;
  }

  hello() {
  console.log('hello' + this.name);
}

}

function Person(name, age) {
  this.name = name;
  this.age = age;
}

Person.prototype.hello = function() {
  console.log('hello' + this.name);
}
